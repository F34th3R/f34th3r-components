## Install

```bash
npm install --save f34th3r-components
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'f34th3r-components'

class Example extends Component {
  render () {
    return (
      <MyComponent />
    )
  }
}
```
